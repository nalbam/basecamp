==== zsh for linux ====
<code>
yum -y install zsh

chsh -s /bin/zsh

</code>

==== zsh for mac ====
<code>
brew install zsh
sudo vim /etc/shells

/usr/local/bin/zsh

chsh -s /usr/local/bin/zsh

chsh -s /bin/zsh

</code>

==== oh-my-zsh ====
<code>
sh -c "$(curl -fsSL https://raw.github.com/robbyrussell/oh-my-zsh/master/tools/install.sh)"

vi .zshrc

ZSH_THEME="dpoggi"

source ~/.nalbam/zshrc.sh

</code>
