		String dt = regdtm;

			SimpleDateFormat f0 = new SimpleDateFormat("yyyyMMddHHmmss");
			SimpleDateFormat f1 = new SimpleDateFormat("yyyy-MM-dd HH:mm");
			Calendar c = Calendar.getInstance();
			c.setTime(f0.parse(dt));
			dt = f1.format(c.getTime());
		}
		catch (Exception e) {
			e.printStackTrace();
		}
		return dt;
	}
